% ── Unicode fallback for glyphs missing in Palatino Linotype ─
% Palatino lacks U+2192 (→), U+2194 (↔), U+FDFA (ﷺ), U+0394 (Δ).
% Map → and ↔ to math-mode versions (always available); use a
% Unicode-coverage fallback font for the Arabic honorific.
\usepackage{newunicodechar}
\newunicodechar{→}{\ensuremath{\rightarrow}}
\newunicodechar{↔}{\ensuremath{\leftrightarrow}}
\newunicodechar{≤}{\ensuremath{\leq}}
\newunicodechar{≥}{\ensuremath{\geq}}
\newunicodechar{Δ}{\ensuremath{\Delta}}
\newfontfamily\arabicfallback{Segoe UI}[Scale=0.95]
\newunicodechar{ﷺ}{{\arabicfallback ﷺ}}

% ── Figures & tables ──────────────────────────────────────────
\usepackage{graphicx}
\usepackage{float}
% Declare `alt` as a no-op graphicx key. Pandoc emits
% \includegraphics[alt={long caption}]{file} and the long alt string
% otherwise routes to dvipdfmx as PDF accessibility metadata that
% overruns its 16-bit page counter on tall figures.
\makeatletter
\define@key{Gin}{alt}{}
\makeatother
% Float placement: htbp lets LaTeX flow figures naturally with the text
% (Nature reading style). intextsep / floatsep / textfloatsep mirror the
% spec-resistance Nature template so figures embed in prose rather than
% landing on standalone pages. The fraction limits below relax LaTeX's
% strict 0.7 / 0.3 defaults so a 3.5-to-4-inch figure can share a page
% with surrounding prose instead of getting punted to a standalone page.
\floatplacement{figure}{!htbp}
\setlength{\intextsep}{3ex plus 1ex minus 0.5ex}
\setlength{\floatsep}{4ex plus 1.5ex minus 0.5ex}
\setlength{\textfloatsep}{3ex plus 1ex minus 0.5ex}
\setcounter{topnumber}{2}
\setcounter{bottomnumber}{2}
\setcounter{totalnumber}{3}
\renewcommand{\topfraction}{0.9}
\renewcommand{\bottomfraction}{0.7}
\renewcommand{\textfraction}{0.1}
\renewcommand{\floatpagefraction}{0.85}
\usepackage{caption}
% Define pipe separator manually (Nature uses "Figure 1 | Caption text")
\DeclareCaptionLabelSeparator{naturepipe}{\enspace\textbar\enspace}
\captionsetup{
  font={footnotesize},
  labelformat=empty,
  justification=justified,
  singlelinecheck=false,
  skip=6pt
}

% Shrink font size for all longtables to help wide tables fit
\usepackage{etoolbox}
\AtBeginEnvironment{longtable}{\footnotesize}
\AtBeginEnvironment{figure}{\setlength{\parindent}{0pt}}

% Allow long monospace strings (e.g. cell IDs like PASHTUN_DAUGHTER_RAN)
% to break at underscores inside longtable cells. Pandoc emits \texttt{...}
% with \_ for each underscore; redefining \_ inside longtable to insert a
% zero-width breakable point lets these wrap rather than overflow into the
% next column.
\AtBeginEnvironment{longtable}{%
  \let\origunderscore\_%
  \def\_{\origunderscore\hspace{0pt}\penalty500\hspace{0pt}}%
}
